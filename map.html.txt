 Real-Time Map Display with Google Maps





<!DOCTYPE html>
<html>
<head>
    <title>Real-Time Team Tracking</title>
    <script src="https://maps.googleapis.com/maps/api/js?key=YOUR_GOOGLE_MAPS_API_KEY&callback=initMap" async defer></script>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/socket.io/4.0.0/socket.io.js"></script>
    <style>
        #map { height: 100vh; width: 100%; }
    </style>
</head>
<body>
    <h3>Maintenance Team Real-Time Tracking</h3>
    <div id="map"></div>
    <script>
        let map;
        const markers = {};

        function initMap() {
            map = new google.maps.Map(document.getElementById('map'), {
                center: { lat: 0, lng: 0 },
                zoom: 12
            });

            const socket = io.connect(window.location.origin);

            socket.on('location_update', function(data) {
                const { team_member_id, latitude, longitude } = data;
                if (markers[team_member_id]) {
                    markers[team_member_id].setPosition(new google.maps.LatLng(latitude, longitude));
                } else {
                    markers[team_member_id] = new google.maps.Marker({
                        position: { lat: latitude, lng: longitude },
                        map: map,
                        label: 'ðŸ‘·'
                    });
                }
            });
        }
    </script>
</body>
</html>
